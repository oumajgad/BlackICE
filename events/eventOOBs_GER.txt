country_event = {
	id = 400038
	is_triggered_only = yes
	title = "Form the 1. Geb.-Division"
	desc = "EVTDESC400038"
	picture = "1-Gebdiv"
	option = {
		name = "Creation of 1. Gebirgsjäger-Division "
		remove_brigade = "Geb.-Rgt. 99/100"
		remove_brigade = "Geb.-Art.-Rgt. 69"
		remove_brigade = "Gefechtstross-Bat. 54"
		set_country_flag = gebdiv_one
		ai_chance = {
			factor = 100
		}
		OMG = {
			any_country = {
				limit = {
					tag = GER
					has_country_flag = BlackICE
				}
                change_variable = {
					which = IC_days_spent
					value = +300
				}
				energy = -600
				metal = -300
				rare_materials = -150
				load_oob = GER/GER_1_GebirgsDiv_build.txt
			}
		}
		OMG = {
			any_country = {
				limit = {
					tag = GER

					not = {
						has_country_flag = BlackICE
					}
				}
				load_oob = GER/GER_1_GebirgsDiv.txt
			}
		}
	}
	option = {
		name="No need"
		set_country_flag = gebdiv_one_no
		ai_chance = {
			factor = 0
		}
	}
}
