		name = "No Need"  # no

			factor = 0
		}
	}
}
# 1st Guard Division
country_event = {
	id = 15001
	is_triggered_only = yes
	title = "1st Guard Division"
	desc = "EVTDESC15001"
	picture = "ROM_1st_Guard_Division"
	option = {
		name = "Form the 1st Guard Division"
		set_country_flag = ROM_1GD

        #Removing Pre-existing Guard elements
        remove_brigade = "Regimente 1/2 Vanatori Garda"
        remove_brigade = "Regiment 1 Artilerie Garda"
        remove_brigade = "Companie Transport Brigada Garda"

		ai_chance = {
			factor = 100
		}
		OMG = {
            any_country = {
                limit = {
                    tag = ROM
                    has_country_flag = BlackICE
                    has_country_flag = OOB_production
                }
                load_oob = ROM/ROM_1st_Guard_Division_build.txt
            }
        }
        OMG = {
            any_country = {
                limit = {
                    tag = ROM
                    has_country_flag = BlackICE
                    has_country_flag = OOB_spawn
                }
                change_variable = {
                    which = IC_days_spent
                    value = +400
                }
                energy = -800
                metal = -400
                rare_materials = -200
                load_oob = ROM/ROM_1st_Guard_Division.txt
            }
        }
        OMG = {
            #AI
            any_country = {
                limit = {
                    tag = ROM
                    not = {
                        has_country_flag = BlackICE
                    }
                }
                load_oob = ROM/ROM_1st_Guard_Division.txt
            }
        }	}
	option = {
		name = "No need"
		set_country_flag = ROM_1GD_no
		ai_chance = {
			factor = 0
		}
	}
}
