# Bismarck is launched event
province_event = {
	id = 48709

	major = yes

	trigger = {
		tag = GER
		not = { has_country_modifier = Bismarck_spawn_penalty }
		has_country_flag = Bismarck_ship
		not = { has_country_flag = Bismarck_launch }
		not = { has_country_flag = Bismarck_no }
	}

	mean_time_to_happen= {
		days = 1
	}

	title = "Bismarck is Launched!"
	desc = "Bismarck_launchDESC"
	picture = "Bismarck"

	option = {
		name = "What a Fine Ship !"
		set_country_flag = Bismarck_launch
		OMG = {
			any_country = {
				limit = { 
					tag = GER 
					has_country_flag = BlackICE 
					has_country_flag = Bismarck_ship 
					not = { has_country_modifier = Bismarck_spawn_penalty }
				}
				load_oob = GER/GER_Bismarck.txt
			}
		}
	}	
}
# Bismarck lost
country_event = {
	
	id = 60800

	major = yes

	trigger = {
		tag = GER
		war = yes
		has_country_flag = Bismarck_launch
		not = { brigade_exist = "Bismarck (BB)" }
		not = { has_country_modifier = Bismarck_build }
		not = { has_country_flag = Bismarck_loss }
		not = { has_country_flag = Bismarck_no }
	}

	mean_time_to_happen= {
		days = 1
	}

	title = "The Bismarck is Lost !"
	desc = "Enemy action has accounted for this great ship ! Naval morale has slumped with the terrible loss."
	picture = "sinking"

	option = {
		name = "How could this happen ?"
		set_country_flag = Bismarck_loss
		ai_chance = { factor = 100 }
		add_country_modifier = {
			name = "Bismarck_sunk" # 5% Naval org penalty
			duration = 30
		}
	}
	option = {
		name = "No, Bismarck is refitting."
		set_country_flag = Bismarck_build

		add_country_modifier = {
			name = "Bismarck_build"
			duration = 180
		}
	}
}
