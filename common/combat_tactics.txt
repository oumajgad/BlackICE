tactic_tactical_withdrawal = { #counters encirclement

tactic_breakthrough = { #counters ambush
	trigger = {
		is_attacker = yes
		OR = {
			has_armour_unit = yes
			skill_advantage = 2
		}
	}
	
	base = 1
	picture = breakthrough
	
	countered_by = tactic_backhand_blow
	
	movement_speed = 0.5
	attacker = 0.25
	defender = -0.15
}

	trigger = {
		is_attacker = no

		}
	}
	
	base = 0
	picture = counterattack

	attacker = -0.2
	defender = 0.25
}
