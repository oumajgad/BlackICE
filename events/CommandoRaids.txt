#################################################################################################
#
# Marneman's Commando/Special Warfare Missions
#
# These events are another attempt at providing some flavor, choice, and variety into the base
# game. Basically, they work as follows:
#
# The player can choose to build a Special Warfare Training Center, once they have researched the
# Elite Commandos tech. Once it is built, the commando missions are available to them, at least
# once their country is at war.
#
# Periodically, the mission selection event will fire. The frequency of the firing is a function
# of the tech level for Elite Commandos the country has. Once it fires, the player constructs their
# commando mission as follows:
#
# 1) Choose type of mission (sabotage resources, black ops missions, etc.)
# 2) Choose specific mission for the type they've chosen
#
# As much I wanted to add a third step, where the player chose the country to attack, the engine
# simply doesn't make this doable from a dynamic perspective. So, unfortunately, the best we can
# do is target a random country, with whom the current country is at war. An unfortunate side
# effect of using random_country is that the tooltips don't work right, so you don't actually know
# most of the time who or where you are attacking :( So I've gone ahead and concealed the misleading
# tooltips (they spam anyway) with some trickery.
#
# There are defense mechanisms against these attacks. There is a new researchable tech
# called "Installation Security". It has 3 levels of research associated with it. Each level
# decreases the chance that a more severe attack will occur (though it can still happen).
#
# NOTE: There are reward units available for countries who complete commando missions with a
# "resounding success". In order to keep these events fairly easy to read and maintain, the
# reward events are located in a separate file (CommandoRaids-Units.txt).
#
# EVENT IDS (999xxx)
# ------------------
#    000 : Build the Special Warfare Training Center
#    001 : Mission dispatch event
# 002 - 009 : Submenus for selecting missions
# 010 - 089 : Mission events and notifications to target countries
# 090 - 099 : Mission status events (failed, minor success, major success, etc.), Misc...
#
# -Marneman
#

# Initiation event for player - build a Special Warfare Training Center
#################################################################################################
country_event = {
	id = 999000
	trigger = {
		# Need the Elite Commando (SF) tech
		commando_brigade_activation = 1

		not = {
			has_country_modifier = build_special_warfare_center_cooldown
			has_country_flag = special_warfare_training_center
		}
		money = 5000
		supplies = 2500
	}
	mean_time_to_happen = {
		days = 30
	}
	title = "EVTNAMEMARNE9000"
	desc = "EVTDESCMARNE9000"
	picture = "commando-special-warfare-training-center"
	option = {
		name = "EVTOPTAMARNE9000"
		set_country_flag = special_warfare_training_center
		money = -5000
		supplies = -2500
		ai_chance = {
			factor = 90
		}
	}
	option = {
		name = "EVTOPTBMARNE9000"
		ai_chance = {
			factor = 10
		}
		add_country_modifier = {
			name = "build_special_warfare_center_cooldown"
			duration = 180
		}
	}
}
