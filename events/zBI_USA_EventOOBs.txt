# not already on active duty into active service for the duration of the war.
country_event = {
	id = 999692
	fire_only_once = yes
	trigger = {
		tag = USA
		year = 1942
		month = 3
		OR = {
			war_with = JAP
			war_with = GER
		}
	}
	title = "Activation of the Reserves and National Guard"
	desc = "EVTDESCMARNE9692"
	picture = "sleeping-giant"
	option = {
		name = "Call up the boys!"
		OMG = {
			any_country = {
				limit = {
					tag = USA
					has_country_flag = BlackICE
				}
                change_variable = {
					which = IC_days_spent
					value = +9600
				}
				energy = -19200
				metal = -9600
				rare_materials = -4800

				load_oob = USA/1942_reserve_1.txt
			}
		}
		OMG = {
			any_country = {
				limit = {
					tag = USA

					not = {
						has_country_flag = BlackICE
					}
				}
				load_oob = USA/1942_reserve_AI.txt
			}
		}
		manpower = -20
		supplies = -12000
		add_country_modifier = {
			name = "Div_HQ_training_NG"
			duration = 60
		}
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name="No need"
		ai_chance = {
			factor = 0
		}
	}
}

# Major Army Command Reorganization in 1942 - splits Army GHQ into three separate commands
#	- Army Ground Forces (AGF)
#	- Army Air Forces (AAF)
#	- Army Service Forces (ASF)
country_event = {
	id = 930007
	fire_only_once = yes
	trigger = {
		tag = USA
		date = 1942.2.28
		4398 = {
			controller = {
				faction = allies
			}
		}
		# Washington, D.C.
	}
	title = "EVTNAMEMARNE930007"
	desc = "EVTDESCMARNE930007"
	picture = "usa_command_reorg"
	option = {
		name = "Yes, let's form these new commands!"
		supplies = -1500
		money = -1500
		load_oob = USA/usa_command_reorg.txt
		# *TRY* to remove now defunct GHQ units (by brigade names unfortunately)
		remove_brigade = "USAAF GHQ"
		remove_brigade = "USAAF GHQ Leader"
		remove_brigade = "USAAF GHQ AA"
		# Add modifiers which will stay in effect until such time that the units are destroyed
		add_country_modifier = {
			name = AGF_effect
			duration = -1
		}
		add_country_modifier = {
			name = AAF_effect
			duration = -1
		}
		add_country_modifier = {
			name = ASF_effect
			duration = -1
		}
		set_country_flag = army_command_reorg
	}
	option = {
		name = "No need, I have other plans!"
	}
}
