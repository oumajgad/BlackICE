		}
		
		date = 1939.10.27

		manpower = 10
		supplies = 500
	}
	
	title = "EVTNAME33301"
	desc = "EVTDESC33301"
	picture = "FRAevent33301"
	
	option = {
		name = "Yes, let's put this division in the production queue!"

		OMG = {
			any_country = {
				limit = { tag = FRA has_country_flag = BlackICE }
				load_oob = fra_first_moroccan.txt
			}
		}
		
		OMG = {
			any_country = {
				limit = { tag = FRA NOT = { has_country_flag = BlackICE } }
				load_oob = fra_first_moroccan.txt
			}
		}
		
		manpower = -10
		supplies = -500
	
		ai_chance = { factor = 100 }
	}

	option = {

	}
}
